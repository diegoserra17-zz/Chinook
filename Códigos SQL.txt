/*Query 1 - Como está o consumo por país?*/
SELECT c.country, 
  	SUM(i.total) as Total
FROM Invoice i
JOIN Customer c
ON i.customerid = c.customerid
GROUP BY c.country
ORDER BY Total Desc

/*Query 2 - Quais os 10 artistas que mais venderam?*/
SELECT a.name,
         SUM(il.unitprice * il.quantity) as total	 
FROM Artist a
JOIN Album al
ON a.artistid = al.artistid
JOIN Track t
ON t.albumid = al.albumid
JOIN InvoiceLine il
ON t.trackid = il.trackid
GROUP BY a.name
ORDER BY total DESC
LIMIT 10

/*Query 3 - Quais são os 5 principais generos consumidos pelos EUA?*/ 
SELECT g.Name, COUNT(t.TrackId) 
FROM Track t
INNER JOIN Genre g
ON t.GenreId = g.GenreId
INNER JOIN InvoiceLine il 
ON t.TrackId = il.TrackId
INNER JOIN Invoice i
ON il.InvoiceId = i.InvoiceId
WHERE i.BillingCountry = 'USA'
GROUP BY g.GenreId
ORDER BY COUNT(t.TrackId) DESC 
LIMIT 0, 5

/*Query 4 - Quais são os registros de venda por ano?*/
SELECT strftime('%Y',DATE(i.invoicedate)) as ano,
        count(il.Quantity) as unidade_ano
FROM invoice i
JOIN InvoiceLine il
ON i.invoiceid = il.invoiceid
group by ano
